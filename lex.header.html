2 Lexical conventions [lex]

2.8 Header names [lex.header]

header-name:
	< h-char-sequence >
	" q-char-sequence "

h-char-sequence:
	h-char
	h-char-sequence h-char

h-char:
	any member of the source character set except new-line and >

q-char-sequence:
	q-char
	q-char-sequence q-char

q-char:
	any member of the source character set except new-line and "
