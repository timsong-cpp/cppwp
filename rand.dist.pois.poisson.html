29.6.8.4.1 Class template poisson_­distribution [rand.dist.pois.poisson]

1
#
A poisson_­distribution random number distribution produces integer values i≥0 distributed according to the discrete probability function

P(i|μ)=e−μμii!.

The distribution parameter μ is also known as this distribution's mean.

template<class IntType = int>
  class poisson_distribution
  {
  public:
    // types
    using result_type = IntType;
    using param_type  = unspecified;

    // constructors and reset functions
    explicit poisson_distribution(double mean = 1.0);
    explicit poisson_distribution(const param_type& parm);
    void reset();

    // generating functions
    template<class URBG>
      result_type operator()(URBG& g);
    template<class URBG>
      result_type operator()(URBG& g, const param_type& parm);

    // property functions
    double mean() const;
    param_type param() const;
    void param(const param_type& parm);
    result_type min() const;
    result_type max() const;
  };
explicit poisson_distribution(double mean = 1.0);
